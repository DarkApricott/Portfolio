Placing Monsters 

  public void PlaceMonster(Transform _newParent, Vector3 _newPosition)
{
    // Set the position and parent transform of the monster
    transform.position = _newPosition;
    transform.parent = _newParent;

    // Set the monster's layer and player detection mask
    playerLayerMask = LayerMask.GetMask("Player");
    gameObject.layer = monsterLayerMask;

    // Instantiate a new HP bar and set its position relative to the monster
    GameObject _hpBar = Instantiate(hpBarPrefab, GameObject.Find("HPCanvas/Mon_HealthBars").transform);
    _hpBar.transform.position = new Vector2(transform.position.x - 0.2f, transform.position.y + 0.5f); // Update HP bar position
    associatedHPBar = _hpBar.GetComponent<Slider>();
    
    // Update the HP bar to reflect the monster's current health
    UpdateHPBar(Health);
}

private void OnEnable()
{
    // Enable the sprite renderer for the monster's first child object
    transform.GetChild(0).GetComponent<SpriteRenderer>().enabled = true;

    // Ensure the monster is placed on a valid tile
    while (!CheckIfTileHasMonster())
    {
        // Check if the monster is not touching a wall (outside of the map)
        if (!CheckIfTouchWall()) 
        {
            // Set the position to the middle of the map
            transform.position = new Vector3(0.5f, -0.5f, 0);
            
            // Continue checking if the middle tile already has a monster
            while (!CheckIfTileHasMonster())
            {
                // Find a new spawn position if needed
                NewSpawnPosition();
            }

            break;
        }
        else
        {
            // Find a new spawn position if the monster is touching a wall
            NewSpawnPosition();
        }
    }
}

(Saving) Player Stats

	using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using TMPro;

public class Stats
{
    public static float Loot, BaseHP = 100, PlayersLeft, Wave = 1, PlayersKilled, MonstersLost, DamageGiven, DamageReceived, LootWon = Loot;
}

public class UIStatsScript : MonoBehaviour
{
    private static TextMeshProUGUI[] texts;

    private void Awake()
    {
        // Get all TextMeshProUGUI components from the children
        texts = GetComponentsInChildren<TextMeshProUGUI>();

        // Reset stats and update the UI to reflect initial values
        ResetStats();
        UpdateUI();
    }

    /// <summary>
    /// Updates a stat with the given input.
    /// </summary>
    /// <param name="_whichStat"></param>
    /// <param name="_changeWith"></param>
    public static void UpdateStat(ref float _whichStat, float _changeWith)
    {
        // Update the stat value
        _whichStat += _changeWith;

        // Refresh the UI with the updated stats
        UpdateUI();

        // If BaseHP has reached 0, trigger the "You Died" function in the UIManagerScript
        if (_whichStat == Stats.BaseHP && Stats.BaseHP <= 0)
        {
            FindObjectOfType<UIManagerScript>().YouDied();
        }

        // If all players have left, trigger the wave completion and increment the wave count
        if (_whichStat == Stats.PlayersLeft && Stats.PlayersLeft == 0)
        {
            FindObjectOfType<WaveManagerScript>().WaveComplete();
            UpdateStat(ref Stats.Wave, +1);
        }
    }

    /// <summary>
    /// Corrects UI visuals with the current stats.
    /// </summary>
    public static void UpdateUI()
    {
        // Save the current stats into an array for easier UI updating
        float[] _savedStats = { Stats.Loot, Stats.BaseHP, Stats.PlayersLeft, Stats.Wave };

        // Loop through saved stats and update the corresponding UI text elements
        for (int i = 0; i < _savedStats.Length; i++)
        {
            texts[i + 1].text = _savedStats[i].ToString();
        }
    }

    private void ResetStats()
    {
        // Reset all the stats to their default values
        Stats.Loot = 0;
        Stats.BaseHP = 100;
        Stats.PlayersLeft = 0;
        Stats.Wave = 1;
        Stats.PlayersKilled = 0;
        Stats.MonstersLost = 0;
        Stats.DamageGiven = 0;
        Stats.DamageReceived = 0;
        Stats.LootWon = 0;

        // Resume the game's time scale to normal
        Time.timeScale = 1;
    }
}

 using System.Collections;

using UnityEngine;

public class MonsterInfoScript : MonoBehaviour
{
    [Header("Monster Info / Stats")]
    public string[] MonsterInfo = { "Name", "Description", "Speed", "Damage", "Range", "Price" };
    public Sprite MonsterImage;

    private StoreScript updaterScript;

    private void Start()
    {
        updaterScript = FindObjectOfType<StoreScript>();
    }

    /// <summary>
    /// Used in UI Store buttons. Updates store's Monster Info panel.
    /// </summary>
    public void GetScript()
    {
        // Assigns this MonsterInfoScript instance to the SelectedMonsterScript in StoreScript
        updaterScript.SelectedMonsterScript = this;
    }
}




Store Script (Setting monster info / buying)

 void Start()
{
    // Get all Image components and TextMeshProUGUI components from the monster info panel's children
    monsterImage = SF_monsterInfoPanel.GetComponentsInChildren<Image>();
    monsterInfoTexts = SF_monsterInfoPanel.GetComponentsInChildren<TextMeshProUGUI>();
}

public void GetMonsterInfoButtonClicked()
{
    // Check if there is existing monster info, if it matches the currently selected monster,
    // and if the panel is active, toggle the panel off
    if (takenMonsterInfo != null && takenMonsterInfo[0] == SelectedMonsterScript.MonsterInfo[0] && SF_monsterInfoPanel.activeInHierarchy)
    {
        SF_monsterInfoPanel.SetActive(false); // Hide the monster info panel
    }
    else
    {
        // If the selected monster is new, update the info panel with new monster data
        UpdateInfoPanel();
    }

    // Fill the panel with the selected monster's info
    FillMonsterInfo();
}

private void UpdateInfoPanel()
{
    // Store the selected monster's information and image
    takenMonsterInfo = SelectedMonsterScript.MonsterInfo;
    takenMonsterImage = SelectedMonsterScript.MonsterImage;

    // Activate the monster info panel to display new data
    SF_monsterInfoPanel.SetActive(true);
}

private void FillMonsterInfo()
{
    // Loop through each piece of the monster's info and update the UI text components
    for (int i = 0; i < takenMonsterInfo.Length; i++)
    {
        monsterInfoTexts[i].text = takenMonsterInfo[i]; // Set each text field with the corresponding monster info
    }

    // Set the monster's image in the UI
    monsterImage[1].sprite = takenMonsterImage;
}

public void BuyMonster()
{
    // Convert the price (stored as a string) to a float
    float _Price = float.Parse(takenMonsterInfo[5]);

    // Check if the player has enough Loot to purchase the monster
    if (Stats.Loot >= _Price)
    {
        // Change the UI screen to the "PlayUI" after purchase
        FindAnyObjectByType<UIManagerScript>().ChangeUIScreen("PlayUI");

        // Deduct the monster's price from the player's loot
        UIStatsScript.UpdateStat(ref Stats.Loot, -_Price);

        // Load the monster prefab based on its name
        GameObject _enemyPrefab = Resources.Load<GameObject>("Prefabs/" + takenMonsterInfo[0]);

        // Enable the PlaceMonsterScript to allow placing the monster
        FindObjectOfType<PlaceMonsterScript>().enabled = true;

        // Instantiate the monster at the specified location
        Instantiate(_enemyPrefab, SF_placeMonster);
    }
}
